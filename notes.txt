Query and mutation examples
query livros {
  livros(filter: {limit: 20}) {
    id,
    titulo,
    numeroDeSerie,
    precoSugerido,
    dataDePublicacao
  }
}

mutation addLivro {
  addLivro(input: {
    titulo: "menino maluquinho",
        numeroDeSerie: 1,
    precoSugerido: 40,
    dataDePublicacao: "25/03/2019",
    
  }) {
    titulo,
    numeroDeSerie,
    precoSugerido,
    dataDePublicacao
  }
}


query autores {
  autores(filter: { limit: 12 }) {
    nome,
    sobrenome,
    dataNascimento,
  	obras {
      titulo
    }
  }
}

Using fetch to make a request 
fetch("http://localhost:4000/", {
		method: 'POST',
		headers: {
			"Content-Type": "application/json"
		},
		body: JSON.stringify({
			query: "autores { autores(filter: { limit: 12 }) {nome,sobrenome,dataNascimento,obras {titulo}}}"
		}),
	})
	.then(res => res.json())
	.then(res => console.log(res.data));
